快速开始

                npm install @ipoodle/mars-cli -g

                // 创建项目（初始化拉取远程模板）
                oc create $name (eg：oc create my-app)
        
                // 交互式生成物料
                oc addcmt UserList -d src/views/user

                // 直接生成物料
                oc addcmt01 UserList -d src/views/demo
